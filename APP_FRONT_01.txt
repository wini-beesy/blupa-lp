  <script>
    function closeAllOverlays() {
      const sidebar = document.getElementById('sidebar-menu');
      const profile = document.getElementById('profile-screen');
      const overlay = document.getElementById('sidebar-overlay');
      
      sidebar.classList.remove('active');
      profile.classList.remove('active');
      overlay.classList.remove('active');
      overlay.style.opacity = '0';
      setTimeout(() => {
        if (!sidebar.classList.contains('active') && !profile.classList.contains('active')) overlay.style.display = 'none';
      }, 300);
    }

    function toggleProfile() {
      const profile = document.getElementById('profile-screen');
      const isActive = profile.classList.contains('active');
      const overlay = document.getElementById('sidebar-overlay');
      
      if (isActive) {
        profile.classList.remove('active');
        overlay.classList.remove('active');
        overlay.style.opacity = '0';
        setTimeout(() => overlay.style.display = 'none', 300);
      } else {
        overlay.style.display = 'block';
        setTimeout(() => {
          profile.classList.add('active');
          overlay.classList.add('active');
          overlay.style.opacity = '1';
        }, 10);
      }
    }

    function toggleSidebar() {
      const sidebar = document.getElementById('sidebar-menu');
      const overlay = document.getElementById('sidebar-overlay');
      const isActive = sidebar.classList.contains('active');
      
      if (isActive) {
        sidebar.classList.remove('active');
        overlay.classList.remove('active');
        overlay.style.opacity = '0';
        setTimeout(() => {
          if (!sidebar.classList.contains('active')) overlay.style.display = 'none';
        }, 300);
      } else {
        overlay.style.display = 'block';
        setTimeout(() => {
          sidebar.classList.add('active');
          overlay.classList.add('active');
          overlay.style.opacity = '1';
        }, 10);
      }
    }
  </script>
